# Команды для обновления данных на GitHub через CMD

# 1. Проверить статус изменений
git status

# 2. Добавить все измененные файлы
git add .

# ИЛИ добавить конкретные файлы:
# git add ewropg.html server.js

# 3. Создать коммит с описанием изменений
git commit -m "Оптимизация производительности и добавление новых стримеров"

# 4. Отправить изменения на GitHub
git push origin main

# ============================================
# Если нужно принудительно добавить файл из .gitignore:
# git add -f data/streamers.json

# Если есть конфликты при push:
# git pull origin main
# (разрешить конфликты, если есть)
# git push origin main

# Если нужно отменить изменения (ОСТОРОЖНО!):
# git restore <file>
# git restore ewropg.html
